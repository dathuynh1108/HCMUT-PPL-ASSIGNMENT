018package org.apache.bcel.generic;
019
020/**
021 * Denotes an unparameterized instruction to store a value into a local variable,
022 * e.g. ISTORE.
023 *
024 */
025public abstract class StoreInstruction extends LocalVariableInstruction implements PopInstruction {
026
027    /**
028     * Empty constructor needed for Instruction.readInstruction.
029     * Not to be used otherwise.
030     * tag and length are defined in readInstruction and initFromFile, respectively.
031     */
032    StoreInstruction(final short canon_tag, final short c_tag) {
033        super(canon_tag, c_tag);
034    }
035
036
037    /**
038     * @param opcode Instruction opcode
039     * @param c_tag Instruction number for compact version, ASTORE_0, e.g.
040     * @param n local variable index (unsigned short)
041     */
042    protected StoreInstruction(final short opcode, final short c_tag, final int n) {
043        super(opcode, c_tag, n);
044    }
045
046
047    /**
048     * Call corresponding visitor method(s). The order is:
049     * Call visitor methods of implemented interfaces first, then
050     * call methods according to the class hierarchy in descending order,
051     * i.e., the most specific visitXXX() call comes last.
052     *
053     * @param v Visitor object
054     */
055    @Override
056    public void accept( final Visitor v ) {
057        v.visitStackConsumer(this);
058        v.visitPopInstruction(this);
059        v.visitTypedInstruction(this);
060        v.visitLocalVariableInstruction(this);
061        v.visitStoreInstruction(this);
062    }
063}
