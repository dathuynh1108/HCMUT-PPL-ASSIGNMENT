018package org.apache.bcel;
019
020/**
021 * Exception constants.
022 * @since 6.0 (intended to replace the InstructionConstant interface)
023 */
024public final class ExceptionConst {
025
026    /**
027     * The mother of all exceptions
028     */
029    public static final Class<Throwable> THROWABLE = Throwable.class;
030
031    /**
032     * Super class of any run-time exception
033     */
034    public static final Class<RuntimeException> RUNTIME_EXCEPTION = RuntimeException.class;
035
036    /**
037     * Super class of any linking exception (aka Linkage Error)
038     */
039    public static final Class<LinkageError> LINKING_EXCEPTION = LinkageError.class;
040
041    /**
042     * Linking Exceptions
043     */
044    public static final Class<ClassCircularityError> CLASS_CIRCULARITY_ERROR = ClassCircularityError.class;
045    public static final Class<ClassFormatError> CLASS_FORMAT_ERROR = ClassFormatError.class;
046    public static final Class<ExceptionInInitializerError> EXCEPTION_IN_INITIALIZER_ERROR = ExceptionInInitializerError.class;
047    public static final Class<IncompatibleClassChangeError> INCOMPATIBLE_CLASS_CHANGE_ERROR = IncompatibleClassChangeError.class;
048    public static final Class<AbstractMethodError> ABSTRACT_METHOD_ERROR = AbstractMethodError.class;
049    public static final Class<IllegalAccessError> ILLEGAL_ACCESS_ERROR = IllegalAccessError.class;
050    public static final Class<InstantiationError> INSTANTIATION_ERROR = InstantiationError.class;
051    public static final Class<NoSuchFieldError> NO_SUCH_FIELD_ERROR = NoSuchFieldError.class;
052    public static final Class<NoSuchMethodError> NO_SUCH_METHOD_ERROR = NoSuchMethodError.class;
053    public static final Class<NoClassDefFoundError> NO_CLASS_DEF_FOUND_ERROR = NoClassDefFoundError.class;
054    public static final Class<UnsatisfiedLinkError> UNSATISFIED_LINK_ERROR = UnsatisfiedLinkError.class;
055    public static final Class<VerifyError> VERIFY_ERROR = VerifyError.class;
056    /* UnsupportedClassVersionError is new in JDK 1.2 */
057//    public static final Class UnsupportedClassVersionError = UnsupportedClassVersionError.class;
058
059    /**
060     * Run-Time Exceptions
061     */
062    public static final Class<NullPointerException> NULL_POINTER_EXCEPTION = NullPointerException.class;
063    public static final Class<ArrayIndexOutOfBoundsException> ARRAY_INDEX_OUT_OF_BOUNDS_EXCEPTION
064                                                            = ArrayIndexOutOfBoundsException.class;
065    public static final Class<ArithmeticException> ARITHMETIC_EXCEPTION = ArithmeticException.class;
066    public static final Class<NegativeArraySizeException> NEGATIVE_ARRAY_SIZE_EXCEPTION = NegativeArraySizeException.class;
067    public static final Class<ClassCastException> CLASS_CAST_EXCEPTION = ClassCastException.class;
068    public static final Class<IllegalMonitorStateException> ILLEGAL_MONITOR_STATE = IllegalMonitorStateException.class;
069
070    /**
071     * Pre-defined exception arrays according to chapters 5.1-5.4 of the Java Virtual
072     * Machine Specification
073     */
074    private static final Class<?>[] EXCS_CLASS_AND_INTERFACE_RESOLUTION = {
075            NO_CLASS_DEF_FOUND_ERROR, CLASS_FORMAT_ERROR, VERIFY_ERROR, ABSTRACT_METHOD_ERROR,
076            EXCEPTION_IN_INITIALIZER_ERROR, ILLEGAL_ACCESS_ERROR
077    }; // Chapter 5.1
078    private static final Class<?>[] EXCS_FIELD_AND_METHOD_RESOLUTION = {
079            NO_SUCH_FIELD_ERROR, ILLEGAL_ACCESS_ERROR, NO_SUCH_METHOD_ERROR
080    }; // Chapter 5.2
081    private static final Class<?>[] EXCS_INTERFACE_METHOD_RESOLUTION = new Class[0]; // Chapter 5.3 (as below)
082    private static final Class<?>[] EXCS_STRING_RESOLUTION = new Class[0];
083    // Chapter 5.4 (no errors but the ones that _always_ could happen! How stupid.)
084    private static final Class<?>[] EXCS_ARRAY_EXCEPTION = {
085            NULL_POINTER_EXCEPTION, ARRAY_INDEX_OUT_OF_BOUNDS_EXCEPTION
086    };
087
088    /**
089     * Enum corresponding to the various Exception Class arrays,
090     * used by {@link ExceptionConst#createExceptions(EXCS, Class...)}
091     */
092    public enum EXCS {
093        EXCS_CLASS_AND_INTERFACE_RESOLUTION,
094        EXCS_FIELD_AND_METHOD_RESOLUTION,
095        EXCS_INTERFACE_METHOD_RESOLUTION,
096        EXCS_STRING_RESOLUTION,
097        EXCS_ARRAY_EXCEPTION,
098    }
099
100    // helper method to merge exception class arrays
101    private static Class<?>[] mergeExceptions(final Class<?>[] input, final Class<?> ... extraClasses) {
102        final int extraLen = extraClasses == null ? 0 : extraClasses.length;
103        final Class<?>[] excs = new Class<?>[input.length + extraLen];
104        System.arraycopy(input, 0, excs, 0, input.length);
105        if (extraLen > 0) {
106            System.arraycopy(extraClasses, 0, excs, input.length, extraLen);
107        }
108        return excs;
109    }
110
111    /**
112     * Creates a copy of the specified Exception Class array combined with any additional Exception classes.
113     * @param type the basic array type
114     * @param extraClasses additional classes, if any
115     * @return the merged array
116     */
117    public static Class<?>[] createExceptions(final EXCS type, final Class<?> ... extraClasses) {
118        switch (type) {
119        case EXCS_CLASS_AND_INTERFACE_RESOLUTION:
120            return mergeExceptions(EXCS_CLASS_AND_INTERFACE_RESOLUTION, extraClasses);
121        case EXCS_ARRAY_EXCEPTION:
122            return mergeExceptions(EXCS_ARRAY_EXCEPTION, extraClasses);
123        case EXCS_FIELD_AND_METHOD_RESOLUTION:
124            return mergeExceptions(EXCS_FIELD_AND_METHOD_RESOLUTION, extraClasses);
125        case EXCS_INTERFACE_METHOD_RESOLUTION:
126            return mergeExceptions(EXCS_INTERFACE_METHOD_RESOLUTION, extraClasses);
127        case EXCS_STRING_RESOLUTION:
128            return mergeExceptions(EXCS_STRING_RESOLUTION, extraClasses);
129        default:
130            throw new AssertionError("Cannot happen; unexpected enum value: " + type);
131        }
132    }
